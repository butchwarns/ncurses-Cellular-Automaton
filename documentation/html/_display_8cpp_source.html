137 void Display::displayRule (int rule, bool* ruleBits)
138  {
139  // Pad rule number with leading zeros
140  std::stringstream strstr;
141  strstr << std::setw(3) << std::setfill('0') << rule;
142 
143  // Print rule in top left corner
144  std::string ruleStr = "Rule: " + strstr.str();
145  mvwaddstr (automatonWin, 0, 0, ruleStr.c_str());
146  wrefresh (automatonWin);
147 
148  // Print rule bits also in top left corner
149  std::string ruleBitStr =" | ";
150  for (int bit = 7; bit >= 0; bit--)
151  {
152  if (ruleBits [bit])
153  ruleBitStr.push_back ('1');
154  else
155  ruleBitStr.push_back ('0');
156  }
157  ruleBitStr.push_back (' ');
158  waddstr (automatonWin, ruleBitStr.c_str());
159  wrefresh (automatonWin);
160  }

187 void Display::advanceCanvas (bool *state)
188  {
189  int canvasRows = automatonWinRows - 2;
190  int canvasCols = automatonWinCols - 2;
191 
192  // Shift all lines up by one dropping the oldest one
193  for (int row = 0; row < (canvasRows - 1); row++)
194  {
195  for (int cell = 0; cell < canvasCols; cell++)
196  {
197  nextCanvas [(row * canvasCols) + cell] = canvas [((row + 1) * canvasCols) + cell];
198  }
199  }
200 
201  // Add new state in last line of next canvas
202  for (int cell = 0; cell < canvasCols; cell++)
203  {
204  nextCanvas [((canvasRows - 1) * canvasCols) + cell] = state [cell];
205  }
206 
207  // Pointer swap to reuse memory
208  bool* temp = canvas;
209  canvas = nextCanvas;
210  nextCanvas = temp;
211  }

232 int Display::enterRule()
233  {
234  std::string enterStr = "Enter Rule: ";
235 
236  // Print message in top left corner
237  mvwaddstr (automatonWin, 0, 0, enterStr.c_str());
238 
239  // Show the cursor and echo key presses
240  curs_set (1);
241  echo();
242  wrefresh (automatonWin);
243 
244  // Have user enter the rule
245  wmove(automatonWin, 0, 12);
246 
247  char numberStr [3];
248  wgetnstr (automatonWin, numberStr, 3);
249 
250  // Convert to int
251  std::stringstream strstr (numberStr);
252 
253  int rule = 0;
254  strstr >> rule;
255 
256  // Hide cursor again and don't echo key presses
257  curs_set (0);
258  noecho();
259  wrefresh (automatonWin);
260 
261  if (rule >= 0 && rule <= 255)
262  {
263  return rule;
264  }
265  else
266  {
267  return 0;
268  }
