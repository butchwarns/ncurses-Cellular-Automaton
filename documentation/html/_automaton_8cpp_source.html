61 void Automaton::init()
62  {
63  if (!stateRandomized)
64  {
65  // Initialise state with one living cell in the middle
66  for (int cell = 0; cell < stateLength; cell++)
67  {
68  if (cell == stateLength / 2)
69  state [cell] = true;
70  else
71  state [cell] = false;
72  }
73  }
74  else
75  {
76  // Seed uniformly distributed random number generation
77  std::random_device randDevice;
78  std::mt19937 mersenneTwister(randDevice());
79  std::uniform_real_distribution<double> distribution (1.0, 10.0);
80 
81  // Initialise state with all random cells
82  for (int cell = 0; cell < stateLength; cell++)
83  {
84  // Generate random cell
85  int randomValue = distribution (mersenneTwister);
86  if (randomValue > 5)
87  {
88  state [cell] = true;
89  }
90  else {
91  {
92  state [cell] = false;
93  }
94  }
95  }
96  }
97 
98  // Initialise rule bit representation
99  calculateRuleBits();
100  }

128 void Automaton::calculateRuleBits()
129  {
130  for (int bit = 0; bit < 8; bit++)
131  {
132  // Isolate bit with bitmask
133  int masked = rule & (int) pow (2, bit);
134 
135  // Shift bit to LSB, essentially making it boolean
136  int shifted = masked >> bit;
137 
138  // Set bit in heap allocated array
139  ruleBits [bit] = shifted;
140  }
141  }

150 void Automaton::advanceState()
151  {
152  // Iterate over cells in current state
153  for (int cell = 0; cell < stateLength; cell++)
154  {
155  // Find cell neighbourhood wrapping over array bounds
156  neighbourhoodBits [0] = state [(cell - 1 + stateLength) % stateLength];
157  neighbourhoodBits [1] = state [cell];
158  neighbourhoodBits [2] = state [(cell + 1) % stateLength];
159 
160  // Convert neighbourhood to integer representation
161  int neighbourhood = 0;
162  for (int i = 0; i < 3; i++)
163  {
164  int bit = neighbourhoodBits [i] ? 1 : 0;
165  neighbourhood += (bit * (int) pow (2, 2 - i));
166  }
167 
168  // Apply rule using bits in reverse order to determine next state
169  switch (neighbourhood)
170  {
171  case (0):
172  nextState [cell] = ruleBits [0];
173  break;
174  case (1):
175  nextState [cell] = ruleBits [1];
176  break;
177  case (2):
178  nextState [cell] = ruleBits [2];
179  break;
180  case (3):
181  nextState [cell] = ruleBits [3];
182  break;
183  case (4):
184  nextState [cell] = ruleBits [4];
185  break;
186  case (5):
187  nextState [cell] = ruleBits [5];
188  break;
189  case (6):
190  nextState [cell] = ruleBits [6];
191  break;
192  case (7):
193  nextState [cell] = ruleBits [7];
194  break;
195  }
196 
197  }
198 
199  // Next state becomes current state
200  // Pointer swap reuses allocated memory
201  bool* temp = state;
202  state = nextState;
203  nextState = temp;
204  }
