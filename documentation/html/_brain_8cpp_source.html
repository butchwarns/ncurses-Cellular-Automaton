115 void Brain::randomize()
116  {
117  // Seed uniformly distributed rng for random rule
118  std::random_device randDevice;
119  std::mt19937 mersenneTwister(randDevice());
120  std::uniform_real_distribution<double> distribution (0.0, 255.0);
121 
122  int randomRule = distribution (mersenneTwister);
123  ca->setRule (randomRule);
124  }
